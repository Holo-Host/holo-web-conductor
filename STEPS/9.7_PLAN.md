```typescript
import { HostFunctionImpl } from "./base";
import { deserializeFromWasm, serializeResult } from "../serialization";
import { encode } from "@msgpack/msgpack";
import { blake2b } from "blakejs";

// Input from WASM (matches hdk RemoteSignal type)
interface RemoteSignalInput {
  agents: Uint8Array[];  // Target AgentPubKeys (39 bytes each)
  signal: Uint8Array;    // ExternIO payload bytes
}

// Signed signal for transport to gateway
export interface SignedRemoteSignal {
  target_agent: number[];    // AgentPubKey as array (for JSON transport)
  zome_call_params: number[]; // Serialized ZomeCallParams
  signature: number[];        // Ed25519 signature (64 bytes)
}

export const sendRemoteSignal: HostFunctionImpl = (context, inputPtr, inputLen) => {
  const { callContext, instance } = context;

  // Deserialize RemoteSignal input
  const input = deserializeFromWasm(instance, inputPtr, inputLen) as RemoteSignalInput;

  // Get sender info from call context
  const [dnaHash, fromAgent] = callContext.cellId;
  const zomeName = callContext.zome;

  // Generate nonce and expiry (like conductor)
  const nonce = crypto.getRandomValues(new Uint8Array(32));
  const expiresAt = BigInt(Date.now() * 1000) + BigInt(300_000_000); // 5 minutes

  // Build signed signals for each target
  const remoteSignals: SignedRemoteSignal[] = [];

  for (const targetAgent of input.agents) {
    // Build ZomeCallParams structure (matches holochain_zome_types::zome_io::ZomeCallParams)
    const zomeCallParams = {
      provenance: fromAgent,
      cell_id: [dnaHash, targetAgent],
      zome_name: zomeName,
      fn_name: "recv_remote_signal",
      cap_secret: null,
      payload: input.signal,
      nonce: nonce,
      expires_at: expiresAt,
    };

    // Serialize params to msgpack
    const serialized = encode(zomeCallParams);

    // Hash the serialized bytes (blake2b-256)
    const hash = blake2b(serialized, undefined, 32);

    // Sign with Lair (synchronous via callContext.lairClient)
    const signature = callContext.lairClient.signByPubKey(fromAgent, hash);

    remoteSignals.push({
      target_agent: Array.from(targetAgent),
      zome_call_params: Array.from(serialized),
      signature: Array.from(signature),
    });
  }

  // Queue for delivery (like emittedSignals pattern)
  if (!callContext.remoteSignals) {
    callContext.remoteSignals = [];
  }
  callContext.remoteSignals.push(...remoteSignals);

  // Return immediately (fire-and-forget semantics)
  return serializeResult(instance, null);
};
```

```rust
/// Send remote signals to target agents via kitsune2 network.
///
/// For each signal:
/// 1. Look up target agent's URL from peer_store
/// 2. Create WireMessage::remote_signal_evt
/// 3. Send via space.send_notify()
pub async fn send_remote_signals(
    &self,
    dna_hash: &DnaHash,
    signals: Vec<SignedRemoteSignalInput>,
) -> (usize, usize) {
    let mut success_count = 0;
    let mut fail_count = 0;

    // Get space for DNA
    let space = match self.get_or_create_space(dna_hash).await {
        Ok(s) => s,
        Err(e) => {
            warn!(%e, "Failed to get space for remote signal");
            return (0, signals.len());
        }
    };

    for signal in signals {
        // Parse target agent
        let target_agent = match AgentPubKey::from_raw_39(signal.target_agent) {
            Ok(a) => a,
            Err(e) => {
                warn!(%e, "Invalid target agent");
                fail_count += 1;
                continue;
            }
        };

        // Look up peer URL from peer store
        let agent_id = target_agent.to_k2_agent();
        let to_url = match space.peer_store().get(agent_id).await {
            Ok(Some(info)) if info.url.is_some() => info.url.unwrap(),
            _ => {
                debug!(%target_agent, "Target agent not found in peer store");
                fail_count += 1;
                continue;
            }
        };

        // Create wire message
        let extern_io = ExternIO(signal.zome_call_params);
        let signature = Signature::from(signal.signature.try_into().unwrap_or([0u8; 64]));
        let wire_msg = WireMessage::remote_signal_evt(target_agent, extern_io, signature);

        // Encode and send
        let encoded = match WireMessage::encode_batch(&[&wire_msg]) {
            Ok(e) => e,
            Err(e) => {
                warn!(%e, "Failed to encode wire message");
                fail_count += 1;
                continue;
            }
        };

        if let Err(e) = space.send_notify(to_url, encoded).await {
            warn!(%e, "Failed to send remote signal");
            fail_count += 1;
        } else {
            success_count += 1;
        }
    }

    (success_count, fail_count)
}
```

## Key Implementation Details

### ZomeCallParams Structure (must match Holochain)
```rust
// From holochain_zome_types::zome_io
struct ZomeCallParams {
    provenance: AgentPubKey,
    cell_id: CellId,
    zome_name: ZomeName,
    fn_name: FunctionName,
    cap_secret: Option<CapSecret>,
    payload: ExternIO,
    nonce: Nonce256Bits,
    expires_at: Timestamp,
}
```

### Signature Contract
- Sign blake2b-256 hash of msgpack-serialized ZomeCallParams
- Use sender's Ed25519 private key via Lair client
- Signature is 64 bytes

### Fire-and-Forget Semantics
- Host function returns immediately
- Network delivery is asynchronous
- Failures are logged but not reported to caller
